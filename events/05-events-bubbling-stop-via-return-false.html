Распространение событий в javascript

return false не остановит событие

Если обработчики реагируют на фазу всплытия, то при клике на inner сначала сработает handler_inner, затем handler_outer.

Если в обработчике handler_inner вернуть false, то это не остановит событие.

#outer

#inner

Код


var outer = document.getElementById('outer');
var inner = document.getElementById('inner');

var handler_outer = function () {
  alert('outer');
};

var handler_inner = function (e) {
  alert('inner');
  //Это не остановит событие
  return false;
};

//Обработчик сработает при всплытии события
outer.addEventListener('click', handler_outer, false);
inner.addEventListener('click', handler_inner, false);
